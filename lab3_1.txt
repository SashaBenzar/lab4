#include <stdio.h>
#include <math.h>

int main()
{
    int n,i,j,s,p=0;
    float x;
    printf("введіть n=");
    scanf("%d",&n);
    printf("введіть x=");
    scanf("%g",&x);
    for (i = 0; i <= n; i++)
    { 
        for (s = 0, j = 0; j <= i; j++)
        {
        s += x+j;
        p += s;
        }
    }
    printf("p=%d\n", p);
    return 0;
}
